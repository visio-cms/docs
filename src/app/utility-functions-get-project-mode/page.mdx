# getProjectMode
Use the `getProjectMode` function to get the project mode. The project mode can be either `LIVE`, `BUILDER` or `PREVIEW`. This function is useful when you want conditional evoke a function or render a component based on the project mode.

```jsx  

import { getProjectMode } from 'visio-cms-lib/utitls';
import {Block} from 'visio-cms-lib/types';
import classNames from 'classnames';


const Header:Block = () => {
  const isLiveMode = getProjectMode() === 'LIVE';
  return (
    <div className={classNames("h-9 p-2 w-full", {
        'fixed top-0 left-0 z-50': isLiveMode,
    })}>Header component</div>
  )
}

```
